function jitter(v, pct) { return Math.max(0, v + (Math.random() - 0.5) * 2 * pct); }

// ─── SPARKLINE ────────────────────────────────────────────────────
function drawSparkline(canvasId, data, color) {
  const canvas = document.getElementById(canvasId);
  if (!canvas) return;
  const dpr = window.devicePixelRatio || 1;
  const rect = canvas.getBoundingClientRect();
  const W = rect.width || canvas.offsetWidth || 200;
  const H = 40;
  canvas.width  = W * dpr;
  canvas.height = H * dpr;
  canvas.style.width  = W + 'px';
  canvas.style.height = H + 'px';
  const ctx = canvas.getContext('2d');
  ctx.scale(dpr, dpr);

  const min = Math.min(...data) * 0.95;
  const max = Math.max(...data) * 1.05;
  const scaleY = v => H - ((v - min) / (max - min)) * (H - 4) - 2;
  const scaleX = i => (i / (data.length - 1)) * W;

  // Fill
  const grad = ctx.createLinearGradient(0, 0, 0, H);
  grad.addColorStop(0, color + '30');
  grad.addColorStop(1, color + '00');
  ctx.beginPath();
  ctx.moveTo(scaleX(0), H);
  data.forEach((v, i) => ctx.lineTo(scaleX(i), scaleY(v)));
  ctx.lineTo(scaleX(data.length - 1), H);
  ctx.closePath();
  ctx.fillStyle = grad;
  ctx.fill();

  // Line
  ctx.beginPath();
  data.forEach((v, i) => i === 0 ? ctx.moveTo(scaleX(i), scaleY(v)) : ctx.lineTo(scaleX(i), scaleY(v)));
  ctx.strokeStyle = color;
  ctx.lineWidth = 1.5;
  ctx.lineJoin = 'round';
  ctx.stroke();

  // Last point dot
  ctx.beginPath();
  const lx = scaleX(data.length - 1), ly = scaleY(data[data.length - 1]);
  ctx.arc(lx, ly, 3, 0, Math.PI * 2);
  ctx.fillStyle = color;
  ctx.fill();
}

// ─── RENDER FUNCTIONS ──────────────────────────────────────────────
function renderMetrics() {
  const cpu = metrics.cpu, mem = metrics.mem, lat = metrics.lat;

  document.getElementById('cpu-value').innerHTML = cpu.toFixed(1) + '<span class="metric-unit">%</span>';
  document.getElementById('mem-value').innerHTML = mem.toFixed(1) + '<span class="metric-unit">%</span>';
  document.getElementById('lat-value').innerHTML = lat.toFixed(3) + '<span class="metric-unit">s</span>';

  document.getElementById('cpu-bar').style.width = cpu + '%';
  document.getElementById('mem-bar').style.width = mem + '%';
  document.getElementById('lat-bar').style.width = Math.min(lat / 3 * 100, 100) + '%';

  const cpuSt  = cpu < 70 ? ['OK','status-ok'] : cpu < 85 ? ['WARN','status-warn'] : ['CRIT','status-critical'];
  const memSt  = mem < 75 ? ['OK','status-ok'] : mem < 90 ? ['WARN','status-warn'] : ['CRIT','status-critical'];
  const latSt  = lat < 1  ? ['OK','status-ok'] : lat < 2  ? ['WARN','status-warn'] : ['CRIT','status-critical'];
  const setStatus = (id, [label, cls]) => { const el = document.getElementById(id); el.textContent = label; el.className = 'metric-status ' + cls; };
  setStatus('cpu-status', cpuSt);
  setStatus('mem-status', memSt);
  setStatus('lat-status', latSt);
}

function renderHealthScore(score) {
  document.getElementById('health-score').textContent = score;
  const circumference = 2 * Math.PI * 58;
  const offset = circumference - (score / 100) * circumference;
  document.getElementById('score-arc').style.strokeDashoffset = offset;
  document.getElementById('score-arc').style.stroke = score >= 80 ? '#00e5a0' : score >= 60 ? '#f5a623' : '#ff4757';

  const statusMap = [
    [80, 'Systems Nominal', 'All infrastructure metrics within healthy thresholds. AI anomaly detector is actively monitoring with no active alerts.'],
    [60, 'Minor Degradation', 'Some metrics approaching warning levels. AI Watchdog is monitoring closely and ready to trigger incident response.'],
    [0,  'Degraded State',   'Infrastructure health below acceptable thresholds. AI incident response pipeline is active.'],
  ];
  const [, title, desc] = statusMap.find(([min]) => score >= min);
  document.getElementById('health-status').textContent = title;
  document.getElementById('health-desc').textContent   = desc;
}

function renderServices() {
  const container = document.getElementById('services-rows');
  container.innerHTML = SERVICES.map(svc => {

function update() {
  metrics.cpu = jitter(metrics.cpu, 3);
  metrics.mem = jitter(metrics.mem, 2);
  metrics.lat = Math.max(0.05, jitter(metrics.lat, 0.02));

  sparkData.cpu.push(metrics.cpu); sparkData.cpu.shift();
  sparkData.mem.push(metrics.mem); sparkData.mem.shift();
  sparkData.lat.push(metrics.lat); sparkData.lat.shift();

  const score = Math.round(100 - (metrics.cpu * 0.35 + metrics.mem * 0.35 + Math.min(metrics.lat * 10, 100) * 0.30));

  renderMetrics();
  renderHealthScore(score);
  updateTimestamp();

  drawSparkline('cpu-spark', sparkData.cpu, '#00e5a0');
  drawSparkline('mem-spark', sparkData.mem, '#f5a623');
  drawSparkline('lat-spark', sparkData.lat, '#3d9cf5');
}
